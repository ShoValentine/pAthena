thor_camp,252,163,4	script	Apprentice Blacksmith#1	813 ,{ 
		.@eac = eaclass();
      
	  if(#language == 1) {
		set @npc$,"[ ^FF0000 prendiz Ferreiro ^000000 ]";
	  }else{
		set @npc$,"[ ^FF0000 Apprentice Blacksmith ^000000 ]";
	  }
	  
	if(checkquest(62839) == 1) {	
		if( @inventorylist_count > 1 ){
			if(#language == 1) {
				mes "Seu espaço de inventário está ficando cheio, por favor, remova algum item";
			}else{
				mes "Your inventory slot is getting full, please remove some item";
			}
			end;
		}
		if (countitem(1148) >= 1 && countitem(2655) >= 1 && countitem(7095) >= 50 && countitem(7094) >= 20 && countitem(7093) >= 20) {
			delitem 1148, 1;
			delitem 2655, 1;
			delitem 7095, 50;
			delitem 7094, 20;
			delitem 7093, 20;
			mes @npc$;
			if(#language == 1) {
				mes "Aqui está seu Meteor Strike";
			}else{
				mes "Here's your Meteor Strike";
			}
			getitem 16043,1;
			completequest 62839;
			end;
		}else{
			mes @npc$;
			if(#language == 1) {
				mes "Você precisará trazer os seguintes itens para eu forjar o Meteor Strike:";
				mes "1x "+callfunc("itemlink",1148)+"";
				mes "1x "+callfunc("itemlink",2655)+"";
				mes "50x "+callfunc("itemlink",7095)+"";
				mes "20x "+callfunc("itemlink",7094)+"";
				mes "20x "+callfunc("itemlink",7093)+"";
				mes "1.000.000 zeny.";
			}else{
				mes "You will need to bring the following items to forge a Meteor Strike:";
				mes "1x "+callfunc("itemlink",1148)+"";
				mes "1x "+callfunc("itemlink",2655)+"";
				mes "50x "+callfunc("itemlink",7095)+"";
				mes "20x "+callfunc("itemlink",7094)+"";
				mes "20x "+callfunc("itemlink",7093)+"";
				mes "1.000.000 zeny.";
			}
			close;
		}
	}
			  
	  
	if(checkquest(62838) == -1) {
		  mes @npc$;
		  if(#language == 1) {
			 mes "Saudações, aventureiro."; 
			 mes "Estou em busca de pessoas que pensam como eu para me ajudar a realizar minhas ambições.";
		  }else{
			 mes "Greetins, adventurer.";
			 mes "I am in search of like-minded individuals to aid me in realizing my ambitions.";
		  }
		  next;
		  mes @npc$;
		  
		if((eaclass()&EAJ_UPPERMASK) != EAJ_SUPER_NOVICE) {
			if(#language == 1) {
				mes "Se você encontrar alguém que você acha que seria adequado para a tarefa, seria ^0000ffSuper Noviço^000000 se você pudesse direcioná-lo para mim.";
				mes "isso realmente ajudaria esse ^0000ffnovato^000000.";
			}else{
				mes "Should you come across anyone you feel would be suited to the task, it would be ^0000ffSuper Novice^000000 if you could direct them to me.";
				mes "it would really help this ^0000ffNovice^000000 out.";
			}
		  end;
		}
		
		if(BaseLevel < 150) {
			if(#language == 1) {
				mes "Você ainda não está forte o suficiente para embarcar nessa aventura.";
				mes "Volte aqui quando tiver mais forte.";
			}else{
				mes "You're not strong enough to embark on this adventure yet.";
				mes "Come back here when you're stronger.";
			}
			end;	  
		}
		
		if(#language == 1) {
			mes "Eu estou ficando sem materiais, estou cheio de serviços";
			mes "Eu não tenho tempo para procurar por eles";
			mes "Você pode me ajudar ?";
			next;
		}else{
			mes "I'm running out of materials and I'm full of services";
			mes "I don't have time to look for them";
			mes "Can you help me?";
			next;
		}
		
		switch(select("Yes, I'll help.", "No!")){
			case 1:	//Yes, I'll help.
			mes @npc$;
			if(#language == 1) {
				mes "Muito obrigado, estou precisando muitos desses materiais.";
				mes "Quando você me trazer esses materiais vou forjar uma arma e te dar de presente.";
			}else{
				mes "Thank you very much, I'm needing many of these materials.";
				mes "When you bring me these materials, I'll forge a weapon and give you a gift.";
			}
			setquest 62838; //Add Quest Window
			close;
			case 2:	//no.
			mes @npc$;
			if(#language == 1) {
				mes "Obrigado por sua atenção, agora me deixe trabalhar.";
			}else{
				mes "Thank you for your attention, now let me work.";
			}
			end;
		}
	}else if(checkquest(62838) == 1) {
		if (countitem(1005) >= 10 && countitem(28106) >= 2 && countitem(1548) >= 5 && countitem(989) >= 1 && countitem(7808) >= 30) {
			delitem 1005, 10;
			delitem 28106, 2;
			delitem 1548, 5;
			delitem 989, 1;
			delitem 7808, 30;
			erasequest 62838;
			
			
			setquest 62839; //Add Quest Window
			
			mes @npc$;
			if(#language == 1) {
				mes "Muito bom.";
			}else{
				mes "Very good.";
			}
			end;
		}else{
			mes @npc$;
			if(#language == 1) {
				mes "Huum, vejo que ainda você não trouxe meus materiais, só posso da seu presente quando me trouxer todos os materiais";
			}else{
				mes "Huum, I see that you have not yet brought my materials, I can only of your gift when you bring me all the materials";
			}
			end;
		}
		
	}  
}
