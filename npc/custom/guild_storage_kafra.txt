function	script	F_KafGStor	{


    set .@n$,"[^0000FFGuild Storage Manager^000000]";
    set @GID,getcharid(2);

    if (@GID == 0) {
        mes .@n$;
        mes "Sorry, but you're not in a Guild.";
        mes "I can not do anything for you.";
		close2;
		end;
	}
	// Unable to access Guild Storage (Busy)
	if (getarg(0) == 1) {
		if (guildopenstorage() == 1) {
			mes getarg(3);
			mes "I'm sorry but another guild member is using the guild storage";
			mes "right now.  Please wait until that person is finished.";
			close2;
			end;
		}
		guildopenstorage();
		close;
	}
}
