prontera,166,198,3	script	Vending Zone	807,{
	warp "vendzone",50,48;
	end;

	// refresh every 5 minute;
	OnInit:
	OnMinute00:
	OnMinute05:
	OnMinute10:
	OnMinute15:
	OnMinute20:
	OnMinute25:
	OnMinute30:
	OnMinute35:
	OnMinute40:
	OnMinute45:
	OnMinute50:
	OnMinute55:
		.total = 0;
		getobjects(BL_PC,0x2, "vendzone");
		.@str1$ = ".@obj_"+BL_PC+"_name$";
		.@pc = getarraysize(getd(.@str1$));
		for (.@i = 0; .@i < .@pc; .@i++) {
			// debugmes getd(.@str1$);
			if(checkvending(getd(.@str1$)))
			{
				.total++;
			}
		}
		delwaitingroom;
		waitingroom "Venders Count: "+.total+"",0;
	end;
}
