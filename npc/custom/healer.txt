-	script	Healer	-1,{

	.@Price = 2500;	// Zeny required for heal
	.@Buffs = 1;	// Also buff players? (1: yes / 0: no)
	.@Delay = 5;	// Heal delay, in seconds

	if (@HD > gettimetick(2))
		end;
	if (BaseLevel == 175 && .@Price > 0){
		if (Zeny < .@Price)
		{
			specialeffect2 EF_HEAL2;
			percentheal 100,100;
			message strcharinfo(0), "[Healer] Not enough zeny to Buff (Healing is free): " + callfunc("F_InsertComma",.@Price) + " Zeny.";
			end;
		}
		message strcharinfo(0), "[Healer] Lv175 - Buffing Charges " + callfunc("F_InsertComma",.@Price) + " Zeny.";
		Zeny -= .@Price;
	}
	else
		message strcharinfo(0), "[Healer] Heal and Buffs are free for non-full level adventurers.";
	specialeffect2 EF_HEAL2;
	percentheal 100,100;
	if (.@Buffs) {
		specialeffect2 EF_INCAGILITY;
		specialeffect2 EF_BLESSING;
		if(vip_status(1))
		{
			sc_start SC_INCREASEAGI,1200000,10;
			sc_start SC_BLESSING,1200000,10;
		}
		else
		{
			sc_start SC_INCREASEAGI,600000,10;
			sc_start SC_BLESSING,600000,10;
		}
	}
	if (.@Delay)
		@HD = gettimetick(2) + .@Delay;
	end;
}
