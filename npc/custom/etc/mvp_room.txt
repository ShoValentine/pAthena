-	script	PrivateBranchRoom	-1,{
	mes "[Private Branch Room]";
	mes "Please select a private Branch room.";
	if ( getvariableofnpc( .rentcost, "Room Manager" ) )
		mes "The cost to rent a room for "+ getvariableofnpc( .timeout, "Room Manager" ) +" minutes is "+ callfunc("F_InsertComma", getvariableofnpc( .rentcost, "Room Manager" ) ) +" zeny.";
	else
		mes "You can only use the room for only "+ getvariableofnpc( .timeout, "Room Manager" ) +" minutes.";
	mes " ";
	for ( .@i = 1; .@i <= 16; .@i++ )
		if ( getvariableofnpc( .renttime[.@i], "Room Manager" ) )
			mes "Room #"+ .@i +" = "+ .color$[ .type[.@i] ] + .whoinuse$[.@i] +"^000000";
	next;
	.@room = select(
		"MVP ROOM 1 [" + getmapusers("pvp_n_1-3") +"]",
		"MVP ROOM 2 [" + getmapusers("pvp_n_2-3") +"]",
		"MVP ROOM 3 [" + getmapusers("pvp_n_3-3") +"]",
		"MVP ROOM 4 [" + getmapusers("pvp_n_4-3") +"]",
		"MVP ROOM 5 [" + getmapusers("pvp_n_5-3") +"]",
		"MVP ROOM 6 [" + getmapusers("pvp_n_6-3") +"]",
		"MVP ROOM 7 [" + getmapusers("pvp_n_7-3") +"]",
		"MVP ROOM 8 [" + getmapusers("pvp_n_8-3") +"]",
		"MVP ROOM 9 [" + getmapusers("pvp_n_1-5") +"]",
		"MVP ROOM 10 [" + getmapusers("pvp_n_2-5") +"]",
		"MVP ROOM 11 [" + getmapusers("pvp_n_3-5") +"]",
		"MVP ROOM 12 [" + getmapusers("pvp_n_4-5") +"]",
		"MVP ROOM 13 [" + getmapusers("pvp_n_5-5") +"]",
		"MVP ROOM 14 [" + getmapusers("pvp_n_6-5") +"]",
		"MVP ROOM 15 [" + getmapusers("pvp_n_7-5") +"]",
		"MVP ROOM 16 [" + getmapusers("pvp_n_8-5") +"]");

	if ( getvariableofnpc( .renttime[.@room], "Room Manager" ) ) {
		if ( .inuseid[.@room] == getcharid( .type[.@room] ) ) {
			if(.@room <= 8) {
				warp "pvp_n_"+ .@room +"-3", 0,0;
			} else {
				warp "pvp_n_"+ (.@room-8) +"-5", 0,0;
			}
			close;
		} else {
			mes "[Private Branch Room]";
			mes "This room is reserved for ";
			mes .color$[ .type[.@room] ] + .whoinuse$[.@room] +"^000000.";
			mes "Please select another.";
			close;
		}
	}
	mes "[Private Branch Room]";
	mes "Reserve this room for...";
	next;
	set .@type, select( "For my party members", "For my guild members", "For personal account use" );
	if ( !getcharid(.@type) ) {
		mes "[Private Branch Room]";
		mes "You do not own a "+( ( .@type == 1 )? "Party" : "Guild" )+".";
		close;
	}
	else if ( Zeny < getvariableofnpc( .rentcost, "Room Manager" ) ) {
		mes "You don't have enough zeny to rent a room.";
		close;
	}
	else if ( getvariableofnpc( .renttime[.@room], "Room Manager" ) ) {
		mes "[Private Branch Room]";
		mes "I'm sorry, somebody else has already registered this room faster than you.";
		close;
	}
	for ( .@i = 1; .@i <= 16; .@i++ ) {
		if ( ( getvariableofnpc( .renttime[.@i], "Room Manager" ) ) && .@type == .type[.@i] && getcharid(.@type) == .inuseid[.@i] ) {
			mes "[Private Branch Room]";
			mes "You already rented Room#"+ .@i +". Use that room instead.";
			close;
		}
	}
	set .type[.@room], .@type;
	set .inuseid[.@room], getcharid(.@type);
	set .whoinuse$[.@room], strcharinfo( ( .@type == 3 )? 0 : .@type );
	Zeny -= getvariableofnpc( .rentcost, "Room Manager" );

	if(.@room <= 8) {
		warp "pvp_n_"+ .@room +"-3", 0,0;
		killmonsterall "pvp_n_"+ .@room +"-3";
	} else {
		warp "pvp_n_"+ (.@room-8) +"-5", 0,0;
		killmonsterall "pvp_n_"+ (.@room-8) +"-5";
	}
	donpcevent "Room Manager#"+ .@room +"::OnEnterMap";
	close;
OnInit:
	.color$[1] =  "^EE8800"; // party color
	.color$[2] =  "^70CC11"; // guild color
	.color$[3] =  "^0000FF"; // account color
	end;
}

				mes "You don't have enough zeny to summon an MVP.";
				close;
			}
		} else {
			set .@menu, select(.mvpid_menu$) -1;
			mes "[Room Manager]";
		}
		mes "Please get ready.";
		close2;
		if ( Zeny < .mvpcost ) end;
		Zeny -= .mvpcost;
		monster "this", 0, 0, "--ja--", .mvpid[.@menu], 1, strnpcinfo(3)+"::OnMobDead";
		end;
	case 3:
		if ( mobcount( "this", strnpcinfo(3)+"::OnMobDead" ) ) {
			mes "[Room Manager]";
			mes "I cannot summon another mini-boss when there are monsters around.";
			close;
		} else if ( .bosscost ) {
			mes "[Room Manager]";
			mes "The cost to summon a mini-boss is "+ callfunc( "F_InsertComma", .bosscost ) +" zeny.";
			next;
			set .@menu, select(.bossid_menu$) -1;
			mes "[Room Manager]";
			if ( Zeny < .bosscost ) {
				mes "You doesn't have enough zeny to summon a mini-boss.";
				close;
			}
		} else {
			set .@menu, select(.bossid_menu$) -1;
			mes "[Room Manager]";
		}
		mes "Please get ready.";
		close2;
		if ( Zeny < .bosscost ) end;
		Zeny -= .bosscost;
		monster "this", 0, 0, "--ja--", .bossid[.@menu], 1, strnpcinfo(3)+"::OnMobDead";
		end;
	case 4:
		if ( mobcount( "this", strnpcinfo(3)+"::OnMobDead" ) > 0 ) {
			mes "[Room Manager]";
			mes "I cannot offer this service when there are monsters around.";
			close;
		}
		close2;
		callshop "MVP room#branch", 1;
		end;
	case 5:
		mes "[Room Manager]";
		mes "Are you sure you want to leave this room?";
		next;
		if ( select( "Yes:No") == 1 )
			warp .respawnmap$, .respawnx, .respawny;
		close;
	case 6:
		mes "[Room Manager]";
		if ( getmapusers( strcharinfo(3) ) > 1 ) {
			mes "There are still some players in this room. Make sure you are the last member in this room to use this option.";
			close;
		}
		mes "Are you sure you want to give up this room?";
		if ( .rentcost )
			mes "You will need to pay again to enter this room.";
		next;
		if ( select( "Yes", "No" ) == 2 ) close;
		awake strnpcinfo(0);
		end;
	}
	close;

OnMobDead:
	end;
