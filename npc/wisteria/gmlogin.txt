-	script	gmsys	-1,{
    
OnPCLoginEvent:
    if(getgmlevel() > 20 ) {
		atcommand "@speed 0";
		//atcommand "@hide";
		atcommand "@monsterignore";
		atcommand "@uptime";
		atcommand "@warp new_zone04 100 108";
		dispbottom "Welcome GM! Please make sure to check players!";
		dispbottom "Current Server Time : " + gettimestr("%A %B %d %Y - %I:%M %p",40);
		end;
    }
    end;
}
