-	script	DeadEvent	-1,{
	end;
	
OnPCDieEvent:
	.@dead_map_name$ = strcharinfo(3);
	
	if((.@dead_map_name$ == "pvp_y_1-1")
			|| (.@dead_map_name$ == "pvp_y_1-4")
			|| (.@dead_map_name$ == "force_1-1")
			|| (.@dead_map_name$ == "force_1-2")
			|| (.@dead_map_name$ == "force_1-3")
			|| (.@dead_map_name$ == "force_2-3")
			|| (.@dead_map_name$ == "prontera")){
		dispbottom "คุณตายใน Map ที่ไม่มีโทษใด ๆ";
		end;
	}
	else{
		exp_multiplier = 0;
		drop_multiplier = 0;
		sc_end SC_EXPBOOST;
		sc_end SC_ITEMBOOST;
		
		// Wish Points
		if(.@dead_map_name$ == instance_mapname("poring_w01",ultima_wish_id)){
			.@temporary_wish_points = wish_points;
			.@temporary_wish_points /= 10;
			if(.@temporary_wish_points <= 0)
			.@temporary_wish_points = 1;
			
			wish_points -= .@temporary_wish_points;
			if(wish_points <= 0)
			wish_points = 0;
			
			dispbottom "แต้มปราถนาลดลง " + F_InsertComma(.@temporary_wish_points) + " คงเหลือ " + F_InsertComma(wish_points);
		}
		
		// Auto warp
		if((.@dead_map_name$ == "bossnia_01")
				|| (.@dead_map_name$ == "bossnia_02")
				|| (.@dead_map_name$ == "bossnia_03")
				|| (.@dead_map_name$ == "bossnia_04")
				|| (.@dead_map_name$ == "tra_fild")
				|| (.@dead_map_name$ == "lasa_sea")
				|| (.@dead_map_name$ == "8@thts")
				|| (.@dead_map_name$ == "1@halo"))
		addtimer 1000,strnpcinfo(3) + "::OnWarp";
		
		end;
	}
	
	end;
	
OnWarp:
	warp "prontera",156,191;
	end;
}
