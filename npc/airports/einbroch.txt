airport,143,43,5	script	Airport Staff#airport1a::airport1	90,{
	mes "[Airport Staff]";
	mes "Selamat datang di";
	mes "Bandara Einbroch,";
	mes "tempat kami menawarkan nonstop";
	mes "penerbangan ke kota-kota";
	mes "Juno, Lighthalzen dan Hugel.";
	next;
	if (select("Naik Kapal Udara:Batal") == 1) {
		mes "[Airport Staff]";
		mes "Biaya naik Kapal Udara";
		mes "adalah 1.200 zeny, tetapi jika Anda sudah";
		mes "mendapat Tiket Kapal Udara Gratis,";
		mes "biaya akan dibebaskan. Maukah";
		mes "Anda naik Kapal Udara?";
		next;
		if (select("Iya:Tidak") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "airport",148,51;
				end;
			}
			mes "[Airport Staff]";
			mes "Maaf, tapi kamu tidak";
			mes "memiliki Tiket Gratis untuk";
			mes "Kapal Udara dan kamu tidak punya";
			mes "cukup zeny untuk boarding";
			mes "Kapal Udara. Ingat, itu";
			mes "biaya boarding 1.200 zeny.";
			close;
		}
	}
	mes "[Airport Staff]";
	mes "Terima kasih dan";
	mes "semoga harimu menyenangkan.";
	close;
}

airport,143,49,3	script	Arrival Staff#airport2a::airport2	90,{
	mes "[Arrival Staff]";
	mes "Selamat datang di Bandara Einbroch.";
	mes "Jika Anda datang dari Anda";
	mes "penerbangan, biarkan saya memandu Anda ke";
	mes "terminal utama. Jika tidak, silakan naik ke Kapal Udara untuk berangkat ke";
	mes "Juno, Lighthalzen dan Hugel.";
	next;
	if (select("Keluar ke terminal utama.:Batal.") == 1) {
		mes "[Arrival Staff]";
		mes "Begitu Anda berada di terminal utama, Anda harus membayar biaya lagi";
		mes "untuk menaiki sebuah Kapal Udara. Anda harus";
		mes "hanya keluar jika kota Einbroch";
		mes "adalah tujuan yang Anda tuju.";
		mes "Lanjutkan ke terminal utama?";
		next;
		if (select("Iya:Tidak") == 1) {
			warp "airport",142,40;
			end;
		}
	}
	mes "[Arrival Staff]";
	mes "Baiklah, terima kasih";
	mes "untuk perlindungan Anda";
	mes "adan saya harap Anda punya";
	mes "penerbangan yang menyenangkan~";
	close;
}
