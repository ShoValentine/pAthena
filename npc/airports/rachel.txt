ra_fild12,295,208,0	script	toairplane#rachel	45,1,1,{
	end;
OnTouch:
	mes "Untuk menggunakan Kapal Udara, Anda harus membayar 1.200 zeny atau Tiket Pesawat Gratis.";
	mes "Apakah Anda ingin menggunakan layanan ini?";
	next;
	if (select("Iya:Tidak") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "Maaf, tapi Anda tidak punya cukup uang.";
		mes "Harap diingat, Anda diharuskan membayar 1.200 zeny untuk menggunakan layanan ini.";
		close;
	}
	mes "Terima kasih, silakan datang lagi.";
	close;
}
