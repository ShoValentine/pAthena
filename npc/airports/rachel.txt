ra_fild12,295,208,0	script	toairplane#rachel	45,1,1,{
	end;
OnTouch:
	mes "หากต้องการใช้เรือเหาะ, คุณจะต้องจ่ายค่าธรรมเนียมจำนวน 1,200 เซนี หรือ คุณจะต้องมี Free Airship Ticket เพื่อโดยสารฟรี.";
	mes "ต้องการที่จะใช้บริการเลยไหม ?";
	next;
	if (select("ใช่:ไม่") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "ขอโทษด้วยนะ, แต่ดูเหมือนว่าคุณมีจำนวนเงินไม่เพียงพอต่อค่าบริการ.";
		mes "โปรดจำไว้ว่า, คุณจะต้องจ่ายค่าธรรมเนียมจำนวน 1,200 เซนี เพื่อใช้บริการ.";
		close;
	}
	mes "ไว้เจอกันใหม่.";
	close;
}
